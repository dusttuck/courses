For each of the following valid arguments, complete a Fitch-style natural deduction to prove its validity. Do this by filling in the missing lines, citations, or both.

\problems
\problem{1} (50 points.)
\argument
 A ^ B
 A > C
+++++++
 C ^ B
\endargument
        \answer
        \fitchproof
         1. | A ^ B
         2. | A > C  ..  Goal: C ^ B
            |---
         3. | A      .  ^ Elim: 1
         4. | B      .  ^ Elim: 1
         5. | 
        \endfitchproof
        \endanswer

\problem{2} (25 points.)
\argument
 A ^ B
 A > C
 [C ^ B] > D
+++++++++++++
 D
\endargument
        \answer
        \fitchproof
         1. | A ^ B
         2. | A > C
         3. | [C ^ B] > D  ..  Goal: D
            |---
         4. | 
        \endfitchproof
        \endanswer

\problem{3} (25 points.)
\argument
 [A > B] ^ [C > D]
 A ^ C
 B > E
+++++++++++++
 D ^ E
\endargument
        \answer
        \fitchproof
         1. | [A > B] ^ [C > D]
         2. | A ^ C
         3. | B > E              ..  Goal: D ^ E
            |---
         4. |
        \endfitchproof
        \endanswer

\endproblems
