For each of the following valid arguments, complete a Fitch-style natural deduction to prove its validity.

\quantifiers
\problems
\problem{1} (40 points.)
\argument
 Ax[Happy(x) > Dog(x)]
 Happy(alonzo) 
+++++++++++++++++++++++
 Dog(alonzo) 
\endargument
        \answer
        \fitchproof
         1. | Ax[H(x) > D(x)]
         2. | H(a)             ..  Goal: D(a)
            |---
         3. | 
        \endfitchproof
        \endanswer

\problem{2} (30 points.)
\argument
 Ax[Logician(x) > Happy(x)]
 Logician(ruth)
++++++++++++++++++++++++++++
 Ex Happy(x) 
\endargument
        \answer
        \fitchproof
         1. | Ax[L(x) > H(x)]
         2. | L(r)             ..  Goal: Ex H(x)
            |---
         3. |
        \endfitchproof
        \endanswer

\problem{3} (20 points.)
\argument
 Ax[Dog(x) > Happy(x)]
 Ax*[[Dog(x) ^ Happy(x)] > Barks(x)*]
++++++++++++++++++++++++++++++++++++++
 Dog(ruth) > Barks(ruth)
\endargument
        \answer
        \fitchproof
         1. | Ax[D(x) > H(x)]
         2. | Ax*[[D(x) ^ H(x)] > B(x)*]  ..  Goal: D(r) > B(r)
            |---
         3. | 
        \endfitchproof
        \endanswer
