(100 points.) For each of the following valid arguments, construct a Fitch-style natural deduction to prove its validity.

\problems
\problem{1}
\argument
 A ^ [B > C]
 A > B
\argumentline
 C ^ A
\endargument
\points{30}
	\answer
	\fitchproof
	 1. | A ^ [B > C]
	 2. | A > B        ..  Goal: C ^ A
	    |---
	 3. | 
	\endfitchproof
	\endanswer

\problem{2}
\argument
 R(a) < S(a)
 b=a ^ P(a)
 [R(b) v P(b)] > S(b)
\argumentline
 R(a)
\endargument
\points{22}
	\answer
	\fitchproof
	 1. | R(a) < S(a)
	 2. | b=a ^ P(a)
	 3. | [R(b) v P(b)] > S(b)  ..  Goal: R(a)
	    |---
	 4. | 
	\endfitchproof
	\endanswer

\problem{3}
\argument
 [A ^ B] > C
 B
 [C v D] > [A ^ F]
\argumentline
 A > F
\endargument
\points{18}
	\answer
	\fitchproof
	 1. | [A ^ B] > C
	 2. | B
	 3. | [C v D] > [A ^ F]  ..  Goal: A > F
	    |---
	 4. | 
	\endfitchproof
	\endanswer

\problem{4}
\argument
 [A ^ B] v [B ^ C]
 A > D
 [B ^ D] > C
\argumentline
 C
\endargument
\points{14}
	\answer
	\fitchproof
	 1. | [A ^ B] v [B ^ C]
	 2. | A > D
	 3. | [B ^ D] > C        ..  Goal: C
	    |---
	 4. | 
	\endfitchproof
	\endanswer

	 4. |

	\endanswer
