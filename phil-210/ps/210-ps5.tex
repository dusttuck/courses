For each of the following valid arguments, complete a Fitch-style natural deduction to prove its validity. Do this by filling in the missing lines, citations, or both.

\problems
\problem{1}
\argument
 A ^ B
 C ^ D
+++++++
 A ^ D
\endargument
        \answer
        \fitchproof
         1. | A ^ B
         2. | C ^ D  ..  Goal: A ^ D
            |---
         3. | A      .  ^ Elim: 1
         4. | B      .  ^ Elim: 1  .  (optional)
         5. |        .  ^ Elim: 2  .  (optional)
         6. |        .  ^ Elim: 2
         7. | A ^ D  .  ^ Intro: 3, 6
        \endfitchproof
        \endanswer

\problem{2}
\argument
 A
 A > B
 B > C
+++++++
 C
\endargument
        \answer
        \fitchproof
         1. | A
         2. | A > B
         3. | B > C  ..  Goal: C
            |---
         4. | B      .  > Elim: 1, 2
         5. | C      .  
        \endfitchproof
        \endanswer

\problem{3}
\argument
 [A v C] ^ [B > D]
 [A v C] > B
+++++++++++++++++++
 D
\endargument
        \answer
        \fitchproof
         1. | [A v C] ^ [B > D]
         2. | [A v C] > B        ..  Goal: D
            |---
         3. | A v C              .  ^ Elim: 1
         4. |                    .  ^ Elim: 1
         5. |                    .  > Elim: 2, 3
         6. | 
        \endfitchproof
        \endanswer

\problem{4}
\argument
 [A > B] ^ C
 [C ^ B] > D
 A ^ E
+++++++++++++
 E ^ D
\endargument
        \answer
        \fitchproof
         1. | [A > B] ^ C
         2. | [C ^ B] > D
         3. | A ^ E        ..  Goal: E ^ D
            |---
         4. | 
        \endfitchproof
        \endanswer

\endproblems
